如 N:=4
10 11 12 1
9 16 13 2
8 15 14 3
7 6 5 4

N =5

12 23 24 17 2

  j:Integer; 
  n，i，k，x，y:Byte;
  result:array[1.. 25,1.. 25] of Byte;
begin
