program xx07_1;
const maxn=1000000;
var n,k,sum:longint;
    f:array[1..maxn] of 0..1;
begin
   readln(n);
   for k:=1 to n do f[k]:=1;
   for k:=1 to n div 2 do f[k*2]:=0;
   for k:=1 to n div 3 do f[k*3]:=0;
   sum:=0;
   for k:=1 to n do sum:=sum+f[k];
   writeln('sum=',sum);
end.
