{
	int * q = (int*)p;
	int n = *q;
	printf("%d\n",n*n);
}
void PrintChar(void * p) {
	char * q = (char*)p;
	printf("%c\n",*q);
}
int main()
{
	int a[5] = {1,2,3,4,5};
