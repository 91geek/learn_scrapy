uses crt;
var
  t:integer;
  i,j,l:byte;
begin
  cursoroff;
  for i:=11 to 22 do begin  
    clrscr;
    textcolor(yellow);
    for j:=1 to 80 do begin
      gotoxy(j,25);         
      write('-');
    end;
    textcolor(green);
    l:=5+(i-10)*3;
    gotoxy(l,i);
    write('\ ');
    gotoxy(l-2,i+1);
    write('>====-');
    gotoxy(l,i+2);
    write('/ ');
    delay(100);
  end;
  t:=100;
  for i:=44 to 74 do begin
    clrscr;
    textcolor(yellow);
    for j:=1 to 80 do begin
      gotoxy(j,25);
      write('-');
    end;
    textcolor(green);
    gotoxy(i,22);
    write('\');
    gotoxy(i-2,23);
    write('>====-');
    gotoxy(i,24);
    write('/');
    delay(t);
    t:=t+10;
  end;
  readln;
end.
