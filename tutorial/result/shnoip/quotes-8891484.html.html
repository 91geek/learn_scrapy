AHOI2001 彩票摇奖 洛谷P2550
（感谢1604还嘉昊同学提供）

#include<iostream>
using namespace std;
short n,i,j,t,s,ans[7];
bool zj[34];
int main()
{
  cin>>n;  //读入小明购买的彩票张数
  for (i=0;i<7;i++) {cin>>t;zj[t]=1;}  //读入中奖号码，并标记zj数组
  for (i=0;i<n;i++) {
    s=0;  //s用于统计彩票中与中奖号码相同的数字个数
    for (j=0;j<7;j++) {
      cin>>t;
      if (zj[t]) s++;
    }
    ans[7-s]++;  //ans用于统计和记录中奖情况，逆序存储
  }
  for (i=0;i<7;i++) cout<<ans[i]<<' ';
  return 0;
}

 

SHOI2002 滑雪 洛谷P1434
（感谢1604陈韬宇同学提供）

//最长上升子序列，将二维降至一维
#include<cstdio>
#include<algorithm>
using namespace std;
int r,c,zz,f[10000],zs,ans;  //f[i]存储到第i项的最长上升子序列的长度
struct hx{
  int x,y,h;  //用结构体存储每个点的横、纵坐标及高度
} da[10000];
bool cmp(hx a,hx b) {
  return a.h<b.h;
}
int main()
{
  scanf("%d%d",&r,&c);
  for (int i=0;i<r;i++)
    for (int j=0;j<c;j++) {
      scanf("%d",&da[zz].h);
      da[zz].x=i;
      da[zz].y=j;
      zz++;
    }
  zs=r*c;
  sort(da,da+zs,cmp);

  f[0]=1;  //开始最长上升子序列处理
  for (int i=1;i<zs;i++) {
    for (int j=0;j<i;j++)
      if (abs(da[i].x-da[j].x)+abs(da[i].y-da[j].y)==1 && da[j].h<da[i].h && f[j]>f[i]) f[i]=f[j];
    f[i]++;
  }

  for (int i=0;i<zs;i++)
    if (f[i]>ans) ans=f[i];
  printf("%d",ans);
  return 0;
}

 

SDOI2005 区间 洛谷P2434

（感谢1604陈韬宇同学提供）


分析：先将所有数据按照上界进行从小到大的排序 然后扫描所有区间,如果两个区间有重叠的区域，就不断更新下界，如果没有重叠了，就输出当前的上下界

#include<bits/stdc++.h>
using namespace std;
struct qj{
int h,t;
}a[50001]; //h和t分别代表每个区间的上下界

bool cmp(qj x,qj y){
return x.h<y.h;
}

int ans1,ans2,n; //ans1,ans2分别为所输出的区间的上下界

int main() {
cin>>n;
for(int i=0;i<n;i++) cin>>a[i].h>>a[i].t;
sort(a,a+n,cmp);

ans1=a[0].h;
ans2=a[0].t; //初始化区间上下界
for(int i=1;i<n;i++){
if(a[i].h<=ans2) ans2=max(ans2,a[i].t); //如果重叠，比较下界并更新
else {
cout<<ans1<<' '<<ans2<<endl;
ans1=a[i].h;
ans2=a[i].t;
} //如果不重叠，就输出并且定义下一个区间的上下界
}
cout<<ans1<<' '<<ans2<<endl; //输出最后一个区间
return 0;
}

 

 HNOI2006 鬼谷子的钱袋 洛谷P2320

（感谢1605张煜东霖同学提供）

分治思想：
假如m=20，可以装一个10金币的钱袋，变成m=10的问题，m=10的问题解决1到10的表示，再一个10金币的钱袋就能表示11到20的问题；以此类推……

#include<cstdio>
using namespace std;
int m,h,jbs[40];
int main()
{
  scanf("%d",&m);
  while (m>0) {
    if (m%2) jbs[h]=m/2+1;
    else jbs[h]=m/2;
    m-=jbs[h];
    h++;
  }
  printf("%d\n",h);
  for (int i=h-1;i>=0;i--) printf("%d ",jbs[i]);
  return 0;
}

 

TJOI2011 序列 洛谷P1732
（感谢1604陈韬宇同学提供）

先来看一下样例：
a 2 9 7 4 6 2 6
b 2 7 2 2 1 0 0

解法一：30分，后面7组测资超时
#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
int t,n,a[100000],ans,zxz,s;
bool b[65537];
int main()
{
  scanf("%d",&t);
  for (int k=0;k<t;k++) {
    scanf("%d",&n);
    memset(b,false,sizeof(b));
    ans=0;
    scanf("%d",&a[0]);ans+=a[0];b[a[0]]=true;
    for (int i=1;i<n;i++) {
      scanf("%d",&a[i]);
      zxz=65537;
      for (int j=0;j<i;j++) {
        s=abs(a[i]-a[j]);
        if (s<zxz) zxz=s;
      }
      b[a[i]]=true;
      ans+=zxz;
    }
    printf("%d\n",ans);
  }
  return 0;
}

解法二：100分
#include<cstdio>
#include<cstring>
using namespace std;
int t,n,a[100000],ans,s;
bool b[65537];
int main()
{
  scanf("%d",&t);
  for (int k=0;k<t;k++) {
    scanf("%d",&n);
    memset(b,false,sizeof(b));
    ans=0;
    scanf("%d",&a[0]);ans+=a[0];b[a[0]]=true;
    for (int i=1;i<n;i++) {
      scanf("%d",&a[i]);
      for (int zxz=0;zxz<=65536;zxz++)  //zxz为当前项和之前项的差值，即min(a[i]-a[j])，从小到大枚举校验
        if (a[i]-zxz>=0 && b[a[i]-zxz] || a[i]+zxz<=65536 && b[a[i]+zxz])  {
          b[a[i]]=true;
          ans+=zxz;
          break;
        }
    }
    printf("%d\n",ans);
  }
  return 0;
}

 

GDOI2014 世界杯 洛谷P3887
解法一：
（感谢1604陈韬宇同学提供）
标签：优先队列
分析：根据题意，每次从仍未选择的球员中选择综合水平值最高的，且选过的人不能再选；故可以采用优先队列来存储和模拟。

#include<bits/stdc++.h>
using namespace std;
priority_queue<int> k,m,d,f;
int k1,d1,m1,f1,t,q,a,b,c;
double sum;
int main()
{
  cin>>k1>>d1>>m1>>f1;
  for(int i=0;i<k1;i++){
    cin>>t;
    k.push(t);
  }
  for(int i=0;i<d1;i++){
    cin>>t;
    d.push(t);
  }
  for(int i=0;i<m1;i++){
    cin>>t;
    m.push(t);
  }
  for(int i=0;i<f1;i++){
    cin>>t;
    f.push(t);
  }

  cin>>q;
  for(int i=0;i<q;i++){
    cin>>a>>b>>c;
    sum=0;
    sum+=k.top();
    k.pop();
    for(int j=0;j<a;j++){
      sum+=d.top();
       d.pop();
    }
    for(int j=0;j<b;j++){
      sum+=m.top();
      m.pop();
    }
    for(int j=0;j<c;j++){
      sum+=f.top();
      f.pop();
    }
    cout<<fixed<<setprecision(2)<<sum/11<<endl;
  }
  return 0;
}

 

解法二：

（感谢1604还嘉昊同学提供）

#include<cstdio>
#include<algorithm>
using namespace std;
int kp,dp,mp,fp,k[100000],d[100000],m[100000],f[100000],q,a,b,c,w,x,y,z;
double ans;
bool cmp(int a,int b) {
  return a>b;
}
int main()
{
  scanf("%d%d%d%d",&kp,&dp,&mp,&fp);
  for (int i=0;i<kp;i++) scanf("%d",&k[i]);
  for (int i=0;i<dp;i++) scanf("%d",&d[i]);
  for (int i=0;i<mp;i++) scanf("%d",&m[i]);
  for (int i=0;i<fp;i++) scanf("%d",&f[i]);
  sort(k,k+kp,cmp);
  sort(d,d+dp,cmp);
  sort(m,m+mp,cmp);
  sort(f,f+fp,cmp);
  scanf("%d",&q);
  for (int i=0;i<q;i++) {
    ans=k[w];w++;  //守门员
    scanf("%d%d%d",&a,&b,&c);
    for (int i=x;i<x+a;i++) ans+=d[i];  //将剩余的能力最高的后卫加到ans中
    x+=a;  //将已将用过的后卫去掉
    for (int i=y;i<y+b;i++) ans+=m[i];
    y+=b;
    for (int i=z;i<z+c;i++) ans+=f[i];
    z+=c;
    printf("%.2lf\n",ans/11);        
  }
  return 0;
}
