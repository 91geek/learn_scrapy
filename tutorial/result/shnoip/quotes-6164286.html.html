begin

  writeln;

begin
  intro;
  start;
  cursoroff;
  clrscr;
  df:=0; randomize;
  writeln;
  writeln;
  writeln('XX小学一年级期末考试（加法计算）':56);
  writeln;
  writeln; 
  for i:=1 to 10 do begin
    a:=random(100);
    b:=random(100);
    c:=a+b;
    write('第',i:2,'题：',a:2,' + ',b:2,' = ');
    readln(d);
    if d=c then df:=df+10;
  end;
  writeln;
  writeln;
  writeln(na,'，你的分数是：',df,'分');
  writeln;
  readln;
end.
